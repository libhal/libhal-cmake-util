cmake_minimum_required(VERSION 3.15)

# CMake helpers from libhal-cmake-helpers
# Source: https://github.com/libhal/libhal-cmake-helpers
find_package(LibhalCMakeUtil REQUIRED)

libhal_project_init(libhal-actuator)

libhal_add_library(libhal-actuator
    SOURCES
        src/rc_servo.cpp
        src/smart_servo/rmd/drc_v2.cpp
        src/smart_servo/rmd/mc_x_v2.cpp
)

libhal_apply_compile_options(libhal-actuator)

libhal_install_library(libhal-actuator)

libhal_add_tests(libhal-actuator
    TEST_SOURCES
        tests/main.test.cpp
        tests/rc_servo.test.cpp
        tests/smart_servo/rmd/drc.test.cpp
        tests/smart_servo/rmd/mc_x.test.cpp
    PACKAGES libhal-mock
    LINK_LIBRARIES libhal::mock
)
